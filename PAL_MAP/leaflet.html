require('leaflet'); require('leaflet-overpass-layer'); var attr_osm = 'Map data © OpenStreetMap contributors'; var attr_overpass = 'POI via Overpass API'; var osm = new L.TileLayer( 'https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { 'opacity': 0.7, 'attribution': [attr_osm, attr_overpass].join(', ') } ); var map = new L.Map('map') .addLayer(osm) .setView(new L.LatLng(52.265, 10.524), 14); var opl = new L.OverPassLayer({ 'query': '(node({{bbox}})[organic];node({{bbox}})[second_hand];);out qt;', }); map.addLayer(opl);
